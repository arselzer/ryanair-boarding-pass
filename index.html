Boarding Pass Generator

Generate your boarding pass barcode without the airline app

1

Get your boarding pass data

  1. Go to ryanair.com and log in to your booking
  2. Navigate to your trip so you can see your flights
  3. Press F12 to open browser DevTools
  4. Click the Console tab
  5. Copy the code below and paste it into the console, then press Enter
(async()=>{function f(){for(let i=0;i<sessionStorage.length;i++){const k=sessionStorage.key(i);try{const v=JSON.parse(sessionStorage.getItem(k));if(v&&v.sessionToken)return v.sessionToken}catch{}}for(let i=0;i<sessionStorage.length;i++){const k=sessionStorage.key(i),v=sessionStorage.getItem(k);if(/^[a-z0-9]{20,30}$/.test(v))return v}const m=document.cookie.match(/(?:fr-session|ryanair-session)=([^;]+)/);if(m)return m[1];try{const el=document.querySelector('[ng-version]');if(el){const ng=el.__ngContext__||el.__ng_context__;if(ng){const s=JSON.stringify(ng);const t=s.match(/"sessionToken":"([a-z0-9]{20,30})"/);if(t)return t[1]}}}catch{}return null}const t=f();if(!t){console.error('Could not find session token automatically.');console.log('Go to Network tab, find any request to /api/, copy the x-session-token header value.');const manual=prompt('Paste your x-session-token:');if(!manual)return;var token=manual}else{var token=t;console.log('Session token found automatically!')}const r=await fetch('/api/booking/boarding-pass/v6/en-gb/boardingPass',{headers:{'client':'desktop','client-version':'1.13.0','x-session-token':token}});const d=await r.json();if(d.boardingPasses){const bp=d.boardingPasses.map(p=>p.barcode.payload);try{await navigator.clipboard.writeText(bp[0]);console.log('Copied to clipboard!')}catch{}console.log('BOARDING PASS BARCODE(S):');bp.forEach((b,i)=>console.log('Passenger '+(i+1)+': '+b));prompt('Copy this barcode string and paste it on the generator page:',bp[0])}else{console.error('No boarding passes found. Make sure you have checked in.',d)}})()

This code runs in your browser on ryanair.com. It automatically finds your session token from the page, fetches the boarding pass, and shows the barcode string. Nothing is sent to any third party.
